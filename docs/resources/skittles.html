Pivot the data

skittles_long <- skittles |> 
  select(2, 4:8) |> 
  pivot_longer(2:6, names_to = "color", values_to = "candies")

skittles_long

New totals through group

skittles_long |> 
  group_by(name) |> 
  summarise(tot = sum(candies)) |> 
  arrange(tot |> desc())

New average through group

skittles_long |> 
  group_by(color) |> 
  summarize(avg_color = mean(candies)) |> 
  arrange(avg_color |> desc())
